  31@cython.boundscheck(False)
  32@cython.wraparound(False)
  33@cython.initializedcheck(False)
  34def not_normal(
  35    trace: ndarray,
  36    sample_rate: cython.int = 100000,
  37    bounds_window: cython.int = 3,
  38    estimate_cutoff: cython.double = 10.0,
  39    replace_factor: cython.int = 8,
  40    replace_gap: cython.int = 2,
  41    threshold_window: cython.double = 2.0,
  42    z_score: cython.double = None,
  43    vector_results: cython.bint = False,
  44    parallel: cython.bint = False
  45) -> tuple:
  46    """
  47    Anomaly detection and baseline determination for (nano)electrochemical time series data. This function combines the
  48    estimate and iteration steps into a single function. There are no required parameters other than the input trace and
  49    sample rate.
  50
  51    Args:
  52        trace (ndarray): The input signal trace.
  53        sample_rate (int): The sample rate of the trace. Default is 100000.
  54        bounds_window (int): The window size for the median filter used in bounding. Default is 3.
  55        estimate_cutoff (float): Initial estimate for the cutoff frequency. Default is 10.0.
  56        replace_factor (int): Factor for replacing events in the calculation trace. Default is 8.
  57        replace_gap (int): Gap for replacing events in the calculation trace. Default is 2.
  58        threshold_window (float): Window size for threshold calculation. Default is 2.0.
  59        z_score (float, optional): Z-score for event detection. If None, it is calculated based on the trace length.
  60        vector_results (bool): Whether to return vector results, this is very expensive. Default is False.
  61        parallel (bool): Whether to run the iteration in parallel. Default is False.
  62
  63    Returns:
  64        tuple: A tuple containing event coordinates, baseline, and a dictionary of results.
  65    """
  66
  67    # Default to 1 expected outlier per trace (computed on length, of course)
  68    if z_score is None:
  69        z_score = norm.ppf(1 - ((1 / len(trace)) / 2))
  70
  71    # Default 3-point median filter for bounding
  72    filtered_trace = bounds_filter(trace, bounds_window)
  73
  74    # Initial estimate for the cutoff frequency and direction
  75    cutoff, event_direction, estimate_results = initial_estimate(
  76        trace,
  77        filtered_trace,
  78        sample_rate,
  79        estimate_cutoff,
  80        threshold_window,
  81        z_score
  82    )
  83
  84    # Iterate based on the initial estimate
  85    args = dict(
  86        trace=trace,
  87        filtered_trace=filtered_trace,
  88        sample_rate=sample_rate,
  89        cutoff=cutoff,
  90        event_direction=event_direction,
  91        replace_factor=replace_factor,
  92        replace_gap=replace_gap,
  93        threshold_window=threshold_window,
  94        z_score=z_score,
  95        vector_results=vector_results
  96    )
  97    if parallel:
  98        event_coordinates, baseline, iteration_results = parallel_iterate(**args)
  99    else:
 100        event_coordinates, baseline, iteration_results = iterate(**args)
 101
 102    return event_coordinates, baseline, dict(estimate=estimate_results, iteration=iteration_results)

 105@cython.boundscheck(False)
 106@cython.initializedcheck(False)
 107def parallel_iterate(
 108    trace: ndarray,
 109    filtered_trace: ndarray,
 110    sample_rate: cython.int,
 111    cutoff: cython.double,
 112    event_direction: str,
 113    replace_factor: cython.int,
 114    replace_gap: cython.int,
 115    threshold_window: cython.double,
 116    z_score: cython.double,
 117    vector_results: cython.bint = False,
 118    segment_size: Optional[int64] = None
 119):
 120    """
 121    Parallel equivalent to the iterate function.
 122
 123    Args:
 124        trace (ndarray): The input signal trace.
 125        filtered_trace (ndarray): The filtered version of the input trace.
 126        sample_rate (int): The sample rate of the trace.
 127        cutoff (float): The cutoff frequency for the baseline filter.
 128        event_direction (str): The direction of events ('up', 'down', or 'biphasic').
 129        replace_factor (int): Factor for replacing events in the calculation trace.
 130        replace_gap (int): Gap for replacing events in the calculation trace.
 131        threshold_window (float): Window size for threshold calculation.
 132        z_score (float): Z-score for event detection.
 133        vector_results (bool): Whether to return vector results, this is very expensive. Default is False.
 134        segment_size (int, optional): Size of segments for parallel processing. Default is None and will calculate automatically.
 135
 136    Returns:
 137        tuple: A tuple containing event coordinates, baseline, and iteration results.
 138    """
 139
 140    assert trace.shape == filtered_trace.shape
 141
 142    args = dict(trace=trace, filtered_trace=filtered_trace, sample_rate=sample_rate, cutoff=cutoff,
 143                event_direction=event_direction, replace_factor=replace_factor, replace_gap=replace_gap,
 144                threshold_window=threshold_window, z_score=z_score, vector_results=vector_results)
 145
 146    # Default to the same size as the confirmed stopping criterion accuracy or threshold accuracy, whichever is larger
 147    minimum_segment: cython.longlong = int(max([200000, int(sample_rate * threshold_window)]))
 148    if segment_size is None or segment_size < minimum_segment:
 149        segment_size = minimum_segment
 150    if segment_size > len(trace):
 151        segment_size = len(trace)
 152    splits: cython.int = len(trace) // segment_size
 153
 154    # Not worth it to parallelize if the trace is too small
 155    if splits < 5:
 156        return iterate(**args)
 157
 158    pool = Pool(processes=min(cpu_count(), splits))
 159    results = pool.starmap(
 160        partial(
 161            iterate,
 162            sample_rate=sample_rate,
 163            cutoff=cutoff,
 164            event_direction=event_direction,
 165            replace_factor=replace_factor,
 166            replace_gap=replace_gap,
 167            threshold_window=threshold_window,
 168            z_score=z_score,
 169            vector_results=vector_results
 170        ),
 171        zip(array_split(trace, splits), array_split(filtered_trace, splits))
 172    )
 173    pool.close()
 174    pool.join()
 175
 176    # Reduce the results
 177    iteration = Iteration(label=f'Final', args=args, trace=trace, filtered_trace=filtered_trace)
 178    baselines = []
 179    thresholds = []
 180    calculation_traces = []
 181    event_coordinates = []
 182    current_len: cython.size_t = 0
 183    for result in results:
 184        baselines.append(result[-1][-1].baseline)
 185        thresholds.append(result[-1][-1].threshold)
 186        calculation_traces.append(result[-1][-1].calculation_trace)
 187        if result[-1][-1].event_coordinates.size != 0:
 188            result[-1][-1].event_coordinates[:, 0] += current_len
 189            result[-1][-1].event_coordinates[:, 1] += current_len
 190            event_coordinates.append(result[-1][-1].event_coordinates)
 191        current_len += len(result[-1][-1].baseline)
 192
 193    iteration.baseline = concatenate(baselines)
 194    iteration.threshold = concatenate(thresholds)
 195    iteration.calculation_trace = concatenate(calculation_traces)
 196    iteration.event_coordinates = concatenate(event_coordinates)
 197    coordinates_view: cython.longlong[:, ::1] = iteration.event_coordinates
 198    # recompute event stats, trace stats and over extractions
 199    iteration.trace_stats = trace_statistics(iteration.calculation_trace, iteration.baseline, coordinates_view)
 200    iteration.event_stats = event_statistics(coordinates_view)
 201    iteration.event_stats['Over Extractions'], iteration.event_stats['Significant Events'] = expected_outliers(
 202        trace,
 203        iteration.baseline,
 204        iteration.threshold,
 205        coordinates_view,
 206        z_score
 207    )
 208
 209    return iteration.event_coordinates, iteration.baseline, iteration

 212@cython.boundscheck(False)
 213@cython.initializedcheck(False)
 214def iterate(
 215    trace: ndarray,
 216    filtered_trace: ndarray,
 217    sample_rate: cython.int,
 218    cutoff: cython.double,
 219    event_direction: str,
 220    replace_factor: cython.int,
 221    replace_gap: cython.int,
 222    threshold_window: cython.double,
 223    z_score: cython.double,
 224    vector_results: cython.bint = False
 225) -> ndarray:
 226    """
 227    Perform iterative anomaly detection and baseline determination on the trace.
 228
 229    Args:
 230        trace (ndarray): The input signal trace.
 231        filtered_trace (ndarray): The filtered version of the input trace.
 232        sample_rate (int): The sample rate of the trace.
 233        cutoff (float): The cutoff frequency for the baseline filter.
 234        event_direction (str): The direction of events ('up', 'down', or 'biphasic').
 235        replace_factor (int): Factor for replacing events in the calculation trace.
 236        replace_gap (int): Gap for replacing events in the calculation trace.
 237        threshold_window (float): Window size for threshold calculation.
 238        z_score (float): Z-score for event detection.
 239        vector_results (bool): Whether to return vector results, this is very expensive. Default is False.
 240
 241    Returns:
 242        ndarray: An array containing event coordinates, baseline, and iteration results.
 243    """
 244
 245    assert trace.shape == filtered_trace.shape
 246
 247    args = dict(sample_rate=sample_rate, cutoff=cutoff, event_direction=event_direction, replace_factor=replace_factor,
 248                replace_gap=replace_gap, threshold_window=threshold_window, z_score=z_score)
 249    results = []
 250
 251    # Iterate until the stopping criterion is met
 252    calculation_trace = trace.copy()
 253    i: cython.int = 1
 254    while True:
 255        results.append(Iteration(label=f'Iteration {i}', args=args, trace=trace, filtered_trace=filtered_trace))
 256        # Baseline and threshold on the calculation trace
 257        baseline = baseline_filter(calculation_trace, cutoff, sample_rate)
 258        threshold = thresholder(calculation_trace, baseline, z_score, int(threshold_window * sample_rate),
 259                                results[-2].event_coordinates if i > 1 else None)
 260
 261        # Calculate trace statistics
 262        results[-1].trace_stats = trace_statistics(calculation_trace, baseline,
 263                                                   results[-2].event_coordinates if i > 1 else None)
 264
 265        if vector_results:
 266            results[-1].baseline = baseline
 267            results[-1].threshold = threshold
 268            results[-1].calculation_trace = calculation_trace
 269
 270        # Direction criterion
 271        if i > 1 and (event_direction == 'down' and results[-1].trace_stats['Influence'] > 0 or
 272                      event_direction == 'up' and results[-1].trace_stats['Influence'] < 0):
 273            event_direction = 'biphasic'
 274
 275        # Locate events from the raw trace
 276        results[-1].event_coordinates = locate_events(trace, filtered_trace, baseline, threshold, event_direction)
 277        results[-1].event_coordinates = sort_coordinates(trace, baseline, results[-1].event_coordinates)
 278        # Calculate event statistics
 279        results[-1].event_stats = event_statistics(results[-1].event_coordinates)
 280
 281        # Stopping criterion
 282        if i > 1 and (results[-1].event_stats['Total Event Samples'] in
 283                      [r.event_stats['Total Event Samples'] for r in results[:-1]]):
 284            break
 285
 286        # Replace events in the calculation trace
 287        calculation_trace = event_replacer(calculation_trace, baseline, results[-1].event_coordinates,
 288                                           replace_factor, replace_gap)
 289
 290        i += 1
 291
 292    # Final compute threshold using std as it is more accurate when outliers are no longer present
 293    threshold = thresholder(calculation_trace, baseline, z_score, int(threshold_window * sample_rate),
 294                            results[-1].event_coordinates, method='std')
 295    # Store the final vectors
 296    results.append(Iteration(label='Final', args=args, trace=trace, filtered_trace=filtered_trace, baseline=baseline,
 297                             threshold=threshold, calculation_trace=calculation_trace,
 298                             trace_stats=results[-1].trace_stats))
 299    # Final extraction is biphasic
 300    results[-1].event_coordinates = locate_events(trace, filtered_trace, baseline, threshold, 'biphasic')
 301    results[-1].event_stats = event_statistics(results[-1].event_coordinates)
 302
 303    # Expected number of extractions which are not events
 304    results[-1].event_stats['False Positives'], results[-1].event_stats['Significant Events'] = expected_outliers(
 305        trace,
 306        baseline,
 307        threshold,
 308        results[-1].event_coordinates,
 309        z_score
 310    )
 311
 312    return results[-1].event_coordinates, baseline, results

 315@cython.boundscheck(False)

 318def initial_estimate(
 319        trace: ndarray,
 320        filtered_trace: ndarray,
 321        sample_rate: cython.int,
 322        cutoff: cython.double,
 323        threshold_window: cython.double,
 324        z_score: cython.double,
 325) -> tuple:
 326    """
 327    Provides an initial estimate for the cutoff frequency and event direction for the iterate function.
 328    This is an improved version of the conventional event extraction method.
 329
 330    Args:
 331        trace (ndarray): The input signal trace.
 332        filtered_trace (ndarray): The filtered version of the input trace.
 333        sample_rate (int): The sample rate of the trace.
 334        cutoff (float): The initial estimate for the cutoff frequency.
 335        threshold_window (float): The window size for threshold calculation.
 336        z_score (float): The Z-score for event detection.
 337
 338    Returns:
 339        tuple: A tuple containing the maximum cutoff frequency, event direction, and the results of the
 340                initial estimate.
 341    """
 342
 343    assert trace.shape == filtered_trace.shape
 344
 345    results = Iteration(
 346        label='Estimate',
 347        args=dict(sample_rate=sample_rate, cutoff=cutoff, threshold_window=threshold_window, z_score=z_score),
 348        trace=trace,
 349        filtered_trace=filtered_trace,
 350        calculation_trace=trace,
 351    )
 352
 353    # Initial baseline and threshold
 354    trace_view: cython.double[::1] = trace
 355    results.baseline = baseline_filter(trace_view, cutoff, sample_rate)
 356    results.threshold = thresholder(trace, results.baseline, z_score, int(threshold_window * sample_rate))
 357
 358    # Calculate trace statistics
 359    results.trace_stats = trace_statistics(trace, results.baseline)
 360
 361    # Determine the starting direction
 362    results.event_direction = 'up' if results.trace_stats['Influence'] > 0 else 'down'
 363
 364    # First extraction
 365    results.event_coordinates = locate_events(trace, filtered_trace, results.baseline, results.threshold,
 366                                              results.event_direction)
 367    # temp
 368    results.event_coordinates = sort_coordinates(trace, results.baseline, results.event_coordinates)
 369    results.calculation_trace = event_replacer(trace, results.baseline, results.event_coordinates, 8, 2)
 370
 371    results.baseline = baseline_filter(results.calculation_trace, cutoff, sample_rate)
 372    results.threshold = thresholder(results.calculation_trace, results.baseline, z_score, int(threshold_window * sample_rate))
 373    results.event_coordinates = locate_events(trace, filtered_trace, results.baseline, results.threshold,
 374                                              'up' if results.event_direction == 'down' else 'down')
 375    results.event_coordinates = sort_coordinates(trace, results.baseline, results.event_coordinates)
 376    results.calculation_trace = event_replacer(results.calculation_trace, results.baseline, results.event_coordinates, 8, 2)
 377
 378    results.baseline = baseline_filter(results.calculation_trace, cutoff, sample_rate)
 379    results.threshold = thresholder(results.calculation_trace, results.baseline, z_score, int(threshold_window * sample_rate))
 380    results.event_coordinates = locate_events(trace, filtered_trace, results.baseline, results.threshold,
 381                                              'biphasic')
 382
 383    results.event_stats = event_statistics(results.event_coordinates)
 384
 385    # Calculate the required cutoffs
 386    threshold_view: cython.double[::1] = results.threshold
 387    coordinates_view: cython.longlong[:, ::1] = results.event_coordinates
 388    max_cutoffs: cython.double[::1] = calculate_cutoffs(trace, results.baseline, threshold_view, coordinates_view,
 389                                                        cutoff, sample_rate)
 390
 391    # Q1 of the required cutoffs, original needs to be added on top
 392    calculated_cutoff: cython.double
 393    try:
 394        calculated_cutoff = percentile(max_cutoffs, 25)
 395    except IndexError:
 396        calculated_cutoff = cutoff
 397    # Do not want to be above recommended antialiasing cutoff
 398    results.event_stats['Max Cutoff'] = calculated_cutoff if calculated_cutoff < sample_rate // 5 else cutoff
 399
 400    return results.event_stats['Max Cutoff'], results.event_direction, results

 403@cython.boundscheck(False)
 404@cython.wraparound(False)
 405@cython.initializedcheck(False)
 406def expected_outliers(
 407        trace: ndarray,
 408        baseline: ndarray,
 409        threshold: ndarray,
 410        event_coordinates: cython.longlong[:, ::1],
 411        z_score: cython.double
 412) -> tuple:
 413    """
 414    Calculate the expected number of false positives and significant events based on the chosen Z-score and the event
 415    profiles. Essentially, we can remove all event samples from the calculation if they exceed the maximium expected
 416    value for the sample size, presuming a perfectly normal sample. This allows an estimate of the number of false
 417    positive extractions based on the chosen Z-score, allowing capture of events which lie below the LOD if the false
 418    positives can be accounted for.
 419
 420    Args:
 421        trace (ndarray): The input signal trace.
 422        baseline (ndarray): The baseline of the trace.
 423        threshold (ndarray): The threshold for event detection.
 424        event_coordinates (ndarray): The coordinates of detected events.
 425        z_score (float): The Z-score for event detection.
 426
 427    Returns:
 428        int: The number of false positives and significant events.
 429    """
 430
 431    assert trace.shape == baseline.shape == threshold.shape
 432
 433    # Single outlier z_score expectation
 434    max_z: cython.double = norm.ppf(1 - ((1 / len(trace)) / 2))
 435    max_threshold: cython.double[::1] = (threshold / z_score) * max_z
 436    adjusted = trace - baseline
 437
 438    # Expected number of extractions which are part of the baseline
 439    total_length: cython.size_t = 0
 440    significant_events: cython.long = 0
 441    event_id: cython.size_t
 442    event_number: cython.size_t = len(event_coordinates)
 443    for event_id in range(event_number):
 444        if (max(abs(adjusted[event_coordinates[event_id, 0]:event_coordinates[event_id, 1]])) >
 445                max_threshold[event_coordinates[event_id, 1]]):
 446            total_length += event_coordinates[event_id, 1] - event_coordinates[event_id, 0] + 1
 447            significant_events += 1
 448
 449    false_positives: cython.double = (2 * (1 - norm.cdf(z_score))) * (len(trace) - total_length)
 450    return false_positives, significant_events

 453@cython.boundscheck(False)
 454@cython.wraparound(False)
 455@cython.initializedcheck(False)
 456def simple_extractor(
 457    trace: ndarray,
 458    baseline: ndarray,
 459    event_coordinates: cython.longlong[:, ::1],
 460    sample_rate: cython.int
 461) -> list:
 462    """
 463    Extract the events from the trace based on the event coordinates and the baseline. Then, simple features are
 464    calculated for each event.
 465
 466    Args:
 467        trace (ndarray): The input signal trace.
 468        baseline (ndarray): The baseline of the trace.
 469        event_coordinates (ndarray): The coordinates of detected events.
 470        sample_rate (int): The sample rate of the trace.
 471
 472    Returns:
 473        list: A list of dictionaries containing event features.
 474    """
 475
 476    assert trace.shape == baseline.shape
 477
 478    # Baseline adjust the trace
 479    adjusted = trace - baseline
 480    event_id: cython.size_t
 481    event_number: cython.size_t = len(event_coordinates)
 482    events = []
 483    for event_id in range(event_number):
 484        vector = adjusted[event_coordinates[event_id, 0]:event_coordinates[event_id, 1] + 1]
 485        events.append({
 486            'ID': event_id + 1,
 487            'Coordinates': (event_coordinates[event_id, 0], event_coordinates[event_id, 1]),
 488            'Vector': vector,
 489            'Amplitude': max(abs(vector)),
 490            'Duration': len(vector) / sample_rate * 1e3,
 491            'Area': sum(abs(vector)) / sample_rate,
 492            'Direction': 'up' if sum(sign(vector)) > 0 else 'down',
 493            'Outlier': False
 494        })
 495
 496    return events

 526@cython.boundscheck(False)
 527@cython.wraparound(False)
 528@cython.initializedcheck(False)
 529def trace_statistics(
 530    trace: ndarray,
 531    baseline: ndarray,
 532    event_coordinates: Optional[cython.longlong[:, ::1]] = None
 533) -> dict:
 534    """
 535    Calculate statistics for the trace.
 536
 537    Args:
 538        trace (ndarray): The input signal trace.
 539        baseline (ndarray): The baseline of the trace.
 540        event_coordinates (ndarray, optional): The coordinates of detected events. Default is None.
 541
 542    Returns:
 543        dict: A dictionary containing trace statistics.
 544    """
 545
 546    assert trace.shape == baseline.shape
 547
 548    results = dict()
 549    # Baseline adjust the trace
 550    adjusted = trace - baseline
 551    # Masking events from the stat calculations
 552    event_id: cython.size_t
 553    event_number: cython.size_t
 554    if event_coordinates is not None:
 555        event_number = len(event_coordinates)
 556    event_mask = zeros(trace.shape, dtype=int32)
 557    event_mask_view: cython.long[::1] = event_mask
 558    if event_coordinates is not None:
 559        for event_id in range(event_number):
 560            event_mask_view[event_coordinates[event_id, 0]:event_coordinates[event_id, 1] + 1] = 1
 561
 562    adjusted: cython.double[::1] = adjusted[where(event_mask == 0)]
 563    # Warnings from Shapiro p-value calculation (not needed)
 564    with catch_warnings():
 565        filterwarnings('ignore')
 566        simplefilter('ignore')
 567        #results['Overall Statistic'] = shapiro(adjusted).statistic
 568        results['Positive Statistic'] = shapiro(adjusted[where(adjusted > percentile(adjusted, 75))]).statistic
 569        results['Negative Statistic'] = shapiro(adjusted[where(adjusted < percentile(adjusted, 25))]).statistic
 570        results['Influence'] = results['Negative Statistic'] - results['Positive Statistic']
 571    results['Standard Deviation'] = std(adjusted)
 572    results['Mean'] = mean(adjusted)
 573
 574    return results

 580def sort_coordinates(trace: ndarray, baseline: ndarray, event_coordinates: cython.longlong[:, ::1]) -> ndarray:
 581    """
 582    Sort event coordinates based on event duration and amplitude. This is used to remove the largest events first,
 583    which mitigates influence problems during the replacement step.
 584
 585    Args:

 608@cython.initializedcheck(False)
 609def calculate_cutoffs(
 610        trace: ndarray,
 611        baseline: ndarray,
 612        threshold: cython.double[::1],
 613        event_coordinates: cython.longlong[:, ::1],
 614        cutoff: cython.double,
 615        sample_rate: cython.int
 616) -> ndarray:
 617    """
 618    Calculate the maximum cutoff frequency for each event presuming a boxcar event profile. This is used to determine
 619    the maximum cutoff frequency for the baseline filter during the iteration step.
 620
 621    Args:
 622        trace (ndarray): The input signal trace.
 623        baseline (ndarray): The baseline of the trace.
 624        threshold (ndarray): The threshold for event detection.
 625        event_coordinates (ndarray): The coordinates of detected events.
 626        cutoff (float): The cutoff frequency used to determine the baseline.
 627        sample_rate (int): The sample rate of the trace.
 628
 629    Returns:
 630        ndarray: The maximum cutoff frequency for each event.
 631    """
 632
 633    assert trace.shape[0] == baseline.shape[0] == tuple(threshold.shape)[0]
 634
 635    # Baseline adjust the trace
 636    adjusted = trace - baseline
 637    max_cutoffs = zeros(event_coordinates.shape[0], dtype=DTYPE)
 638    max_cutoffs_view: cython.double[::1] = max_cutoffs
 639    # Calculate the cutoff frequency for each event
 640    event_id: cython.size_t
 641    event_number: cython.size_t = len(event_coordinates)
 642    event_start: cython.longlong
 643    event_end: cython.longlong
 644    vector: cython.double[::1]
 645    damping_ratio: cython.double
 646    for event_id in range(event_number):
 647        event_start = event_coordinates[event_id, 0]
 648        event_end = event_coordinates[event_id, 1]
 649        vector = abs(adjusted[event_start:event_end + 1])

 651        # -3dB of boxcar by length
 652        max_cutoffs_view[event_id] = 0.442947 / (sqrt((len(vector) * damping_ratio) ** 2 - 1) * (1 / sample_rate))
 653        max_cutoffs_view[event_id] += cutoff

 676@cython.boundscheck(False)
 677@cython.initializedcheck(False)
 678def event_replacer(
 679        trace: cython.double[::1],
 680        baseline: cython.double[::1],
 681        event_coordinates: cython.longlong[:, ::1],
 682        replace_factor: cython.int,
 683        replace_gap: cython.int
 684) -> ndarray:
 685    """
 686    The heart of the iterative process, this function replaces events in a copy of the trace (calculation trace)
 687    using an event specific filter. This trace is then used in the iterative process to calculate the baseline and
 688    threshold for the next iteration.
 689
 690    Args:
 691        trace (ndarray): The input signal trace.
 692        baseline (ndarray): The baseline of the trace.
 693        event_coordinates (ndarray): The coordinates of detected events.
 694        replace_factor (int): Factor for replacing events in the calculation trace.
 695        replace_gap (int): Gap for replacing events in the calculation trace.
 696
 697    Returns:
 698        ndarray: The trace with events replaced (calculation trace).
 699    """
 700
 701    assert tuple(trace.shape) == tuple(baseline.shape)
 702
 703    if len(event_coordinates) == 0:
 704        return asarray(trace, dtype=DTYPE)
 705
 706    # Replacing with the baseline
 707    event_id: cython.size_t
 708    event_number: cython.size_t = len(event_coordinates)
 709    padded_trace: cython.double[::1] = zeros(trace.shape[0], dtype=DTYPE)
 710    padded_trace[:] = trace
 711    if replace_gap == 0 and replace_factor == 0:
 712        for event_id in range(event_number):
 713            padded_trace[event_coordinates[event_id, 0] + 1:event_coordinates[event_id, 1]] = (
 714                        baseline[event_coordinates[event_id, 0] + 1:event_coordinates[event_id, 1]])
 715        return asarray(padded_trace, dtype=DTYPE)
 716
 717    # Pad now to avoid padding in the loop (largest event)
 718    longest: cython.int = int(ceil(max([event[1] - (event[0] + 1) for event in event_coordinates]) *
 719                              (replace_factor + replace_gap)))
 720    if longest > len(trace):
 721        padded_trace = concatenate((trace[0] * ones(longest - len(trace)), trace[::-1], trace,
 722                                    trace[::-1], trace[-1] * ones(longest - len(trace))))
 723    else:
 724        padded_trace = concatenate((trace[:longest][::-1], trace, trace[-longest:][::-1]))
 725    padded_trace: cython.double[::1] = padded_trace
 726
 727    # We remove the events in order of largest to smallest (see: sort_coordinates)
 728    event_length: cython.longlong
 729    event_start: cython.longlong
 730    event_end: cython.longlong
 731    length: cython.long
 732    window: cython.double[::1]
 733    segment: cython.double[::1]
 734    filtered_segment: cython.double[::1]
 735    for event_id in range(event_number):
 736        # Adjust for padding, with extra sauce
 737        event_length = event_coordinates[event_id, 1] - (event_coordinates[event_id, 0] + 1)
 738        event_start = (event_coordinates[event_id, 0] + 1) + longest
 739        event_end = event_coordinates[event_id, 1] + longest
 740
 741        # Create the filter window
 742        length = int(round(event_length * (replace_factor + replace_gap)))
 743        length = length if length % 2 == 1 else length + 1
 744        # Pointless
 745        if length == 1:
 746            continue
 747        window = ones(length, dtype=DTYPE)
 748        if replace_gap > 0:
 749            window[(event_length * replace_factor) // 2:-((event_length * replace_factor) // 2)] = 0
 750        window = window / sum(window)
 751
 752        # Filter and replace
 753        segment = padded_trace[event_start - (len(window) // 2):event_end + (len(window) // 2)]
 754        filtered_segment = oaconvolve(segment, window, 'same')
 755        padded_trace[event_start:event_end] = filtered_segment[(len(window) // 2):-(len(window) // 2)]
 756
 757    return asarray(padded_trace[longest:-longest], dtype=DTYPE)

 798@cython.boundscheck(False)
 799@cython.wraparound(False)
 800@cython.initializedcheck(False)
 801def event_bounds(threshold_idxs: cython.longlong[::1], baseline_idxs: cython.longlong[::1]) -> ndarray:
 802    """
 803    Determine the bounds of events based on threshold and baseline crossing indices.
 804
 805    Args:
 806        threshold_idxs (ndarray): Indices where the trace crosses the threshold.
 807        baseline_idxs (ndarray): Indices where the trace crosses the baseline, this is a masked version of the trace.
 808
 809    Returns:
 810        ndarray: The coordinates of detected events.
 811    """
 812
 813    spike_start: cython.size_t
 814    spike_end: cython.size_t = 0
 815    i: cython.size_t
 816    baseline_idxs_length: cython.size_t = len(baseline_idxs)
 817    event_coordinates = []
 818    # Iterate through the threshold crossing indices
 819    for i in range(len(threshold_idxs)):
 820        if threshold_idxs[i] < spike_end:
 821            continue
 822
 823        # Iterate backwards to find the start of the event
 824        spike_start = threshold_idxs[i]
 825        while baseline_idxs[spike_start] == 0 and spike_start > 0:
 826            spike_start -= 1
 827
 828        # Iterate forwards to find the end of the event
 829        spike_end = threshold_idxs[i] + 1
 830        while baseline_idxs[spike_end] == 0 and spike_end < baseline_idxs_length - 2:
 831            spike_end += 1
 832
 833        # If the end was already passed the cutoff sample
 834        if spike_end > baseline_idxs_length - 2:
 835            spike_end = baseline_idxs_length - 2
 836
 837        # This is the OTHER side of the baseline to the event, on both start/end
 838        event_coordinates.append((spike_start, spike_end + 1))
 839
 840    return asarray(event_coordinates, dtype=int64)

 843@cython.boundscheck(False)
 844@cython.initializedcheck(False)
 845def baseline_filter(trace: cython.double[::1], cutoff: cython.double, sample_rate: cython.int) -> ndarray:
 846    """
 847    Apply a baseline filter to the trace using a boxcar filter (moving average). This is a low-pass filter, the result
 848    is to be subtracted which is then a high-pass filter.
 849
 850    Args:
 851        trace (ndarray): The input signal trace.
 852        cutoff (float): The cutoff frequency for the baseline filter.
 853        sample_rate (int): The sample rate of the trace.
 854
 855    Returns:
 856        ndarray: The baseline.
 857    """
 858
 859    # Cutoff to boxcar length conversion (odd length)
 860    length: cython.size_t = int(round(sqrt(0.196202 + ((cutoff * (1 / sample_rate)) ** 2)) /
 861                                (cutoff * (1 / sample_rate))))
 862    length = length if length % 2 == 1 else length + 1
 863    # Pointless
 864    if length == 1:
 865        length = 3
 866    # Ensure the window is not longer than the trace
 867    if length >= len(trace):
 868        length = len(trace) - 1
 869    # Odd length window
 870    window = ones(length, dtype=DTYPE) / length
 871    # Add fit for the edge cases
 872    padded_trace: cython.double[::1] = concatenate((trace[:int(length // 2)][::-1], trace,
 873                                                    trace[-int(length // 2):][::-1]))
 874    # Calculate the moving average
 875    baseline = oaconvolve(padded_trace, window, 'same')
 876    # Cut off the edges
 877    baseline = baseline[int(length // 2):-int(length // 2)]
 878
 879    return baseline

 910@cython.boundscheck(False)
 911@cython.wraparound(False)
 912@cython.initializedcheck(False)
 913def thresholder(
 914        trace: ndarray,
 915        baseline: ndarray,
 916        z_score: cython.double,
 917        window: Optional[int64] = None,
 918        event_coordinates: Optional[cython.longlong[:, ::1]] = None,
 919        method: str = 'iqr'
 920) -> ndarray:
 921    """
 922    Calculate the threshold for event detection based on the trace and baseline. The baseline is subtracted from the
 923    trace (HPF) and the Wan et al. (doi:10.1186/1471-2288-14-135) method is used to estimate the standard deviation.
 924
 925    Args:
 926        trace (ndarray): The input signal trace.
 927        baseline (ndarray): The baseline of the trace.
 928        z_score (float): The Z-score for event detection.
 929        window (int, optional): The window size in samples for threshold calculation. Default is None.
 930        event_coordinates (ndarray, optional): The coordinates of detected events. Default is None.
 931        method (str): The method for threshold calculation ('iqr' or 'std'). Default is 'iqr'.
 932
 933    Returns:
 934        ndarray: The calculated threshold.
 935    """
 936
 937    assert trace.shape == baseline.shape
 938
 939    # Baseline adjust the trace
 940    adjusted = trace - baseline
 941    adjusted_view: cython.double[::1] = adjusted
 942
 943    # Masking events from the std calculation
 944    event_id: cython.size_t
 945    event_number: cython.size_t
 946    if event_coordinates is not None:
 947        event_number = len(event_coordinates)
 948    event_mask = zeros(trace.shape, dtype=int32)
 949    event_mask_view: cython.long[::1] = event_mask
 950    if event_coordinates is not None:
 951        for event_id in range(event_number):
 952            event_mask_view[event_coordinates[event_id, 0]:event_coordinates[event_id, 1] + 1] = 1
 953
 954    # Number of segments to divide the trace into
 955    window: cython.longlong = int(round(window)) if window is not None else len(trace)
 956    segments: cython.long = len(trace) // window if len(trace) // window else 1
 957
 958    # Calculate the threshold in segments
 959    i: cython.long
 960    start: cython.longlong
 961    end: cython.longlong
 962    current_segment: cython.double[::1]
 963    q1: cython.double
 964    q3: cython.double
 965    n: cython.longlong
 966    standard_deviation = zeros(trace.shape, dtype=DTYPE)
 967    standard_deviation_view: cython.double[::1] = standard_deviation
 968    for i in range(segments):
 969        start = i * window
 970        end = (i + 1) * window if i != segments - 1 else len(trace)
 971
 972        # Baseline adjusted segment where events are masked
 973        current_segment = adjusted[start:end][where(event_mask[start:end] == 0)]
 974        if len(current_segment) == 0:
 975            current_segment = adjusted_view[start:end]
 976
 977        if method == 'iqr':
 978            # doi:10.1186/1471-2288-14-135
 979            q1 = quantile(current_segment, 0.25)
 980            q3 = quantile(current_segment, 0.75)
 981            n = len(current_segment)
 982            standard_deviation_view[start:end] = (q3 - q1) / (2 * norm.ppf((0.75 * n - 0.125) / (n + 0.25)))
 983        else:
 984            standard_deviation_view[start:end] = std(current_segment)
 985
 986    return standard_deviation * z_score
